Q8 code : Create an input and a button that says "Add". Whatever is typed in the input should get added as a new item to a <ul> below. After adding, the input should clear itself.
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        
    </style>
</head>
<body>
    <input type="text" placeholder="Type Something...." class="input">
    <button class="add">ADD</button>
    <div id="forList"></div>
       
    <script>
        let input = document.querySelector(".input");
        let button = document.querySelector(".add");
        let div = document.querySelector("#forList");
        button.addEventListener("click", () => {
            // storing the input in text
            let text = input.value;
            // if input is empty is cant be added in list 
            if(text === "") return;
            // creating new element
            let li = document.createElement("li");
            li.innerText = text;
            // add task to list
            div.appendChild(li);
            // clear input
            input.value = "";

        })
    </script>
</body>
</html>

Q10 code : Create an input box. When the user types, if the input is empty show a red border. If it has text show a green border. Use classList to add/remove border classes, not inline styles.
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Input Border Validation</title>
    <style>
        .input {
            padding: 10px;
            border: 2px solid black;
            outline: none;
        }
        .green-border {
            border-color: green;
        }
        .red-border {
            border-color: red;
        }
    </style>
</head>
<body>
    <input type="text" placeholder="Type something..." class="input">
    <script>
        let input = document.querySelector(".input");
        input.addEventListener("input", () => {
            let text = input.value;
            if(text.trim() === "") {
                input.classList.remove("green-border");
                input.classList.add("red-border");
            } else {
                input.classList.remove("red-border");
                input.classList.add("green-border");
            }
        });
    </script>
</body>
</html>
